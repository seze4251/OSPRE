  563             apnd = minval;

  565             apnd = b_i + (ndbl - 1.0);
  566           } else {

  568           }

  588               }

  592               } else {

  594                 lnR->data[jj + 1] = apnd - (double)jj;

  596             }
  597           }
  598         }

  601         Ex_chunk->size[0] = lnR->size[1];

  604         Ey_chunk->size[0] = lnR->size[1];
