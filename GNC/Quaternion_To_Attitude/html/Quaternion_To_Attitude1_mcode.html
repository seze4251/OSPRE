1function [r_SC_body] = Quaternion_To_Attitude(q)
 2%%% Calculates the spacecraft attitude unit vector in ECI.
 3%
 4% Inputs:   q   Quaternion  (vector)
 5%
 6% Author:   Cameron Maywood
 7% Created:  9/29/2016
 8% Modified: 9/29/2016
 9%
10%%
11%%% Direct cosine matrix from quaternion
12DCM = [q(1)^2 + q(2)^2 - q(3)^2 - q(4)^2,   2*(q(2)*q(3) - q(4)*q(1)),          2*(q(2)*q(4) + q(3)*q(1));
13       2*(q(2)*q(3) + q(4)*q(1)),           q(1)^2 - q(2)^2 + q(3)^2 - q(4)^2,  2*(q(3)*q(4) - q(2)*q(1));
14       2*(q(2)*q(4) - q(3)*q(1)),           2*(q(3)*q(4) + q(2)*q(1)),          q(1)^2 - q(2)^2 - q(3)^2 + q(4)^2];
15
16%%% Initial ECI frame (assumed to be pointing in the inertial X-axis direction)
17r_initial = [1; 0; 0];
18
19%%% Spacecraft attitude vector in ECI
20r_SC_body = DCM*r_initial;
21
22end
