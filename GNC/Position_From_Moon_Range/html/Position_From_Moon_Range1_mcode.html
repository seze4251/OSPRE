1function [r_E_SC] = Position_From_Moon_Range(r_E_M, q_M, alpha, beta, theta)
 2%%% Calculates the spacecraft position vector in ECI.
 3%
 4% Inputs:   r_E_M   Moon ECI position vector            (km)
 5%           q_M     Spacecraft_Moon quaternion          (vector)
 6%           alpha   Horizontal quaternion error angle   (deg)
 7%           beta    Vertical quaternion error angle     (deg)
 8%           theta   Moon angular diameter               (deg)
 9%
10% Author:   Cameron Maywood
11% Created:  9/29/2016
12% Modified: 9/29/2016
13%
14%%
15%%% Moon radius, km
16R_M = 1738.1
17
18%%% Spacecraft-Moon attitude unit vector
19r_SC_M = Quaternion_To_Attitude(q_M);
20
21%%% Initialize r_M_SC
22r_M_SC = zeros(3,1);
23
24%%% X component of Moon-SC position vector
25r_M_SC(1) = (R_M/tand(theta/2))*(r_SC_M(2)*sind(alpha)-cosd(alpha)*(r_SC_M(1)*cosd(beta)+r_SC_M(3)*sind(beta)));
26
27%%% Y component of Moon-SC position vector
28r_M_SC(2) = (-R_M/tand(theta/2))*(r_SC_M(2)*cosd(alpha)+sind(alpha)*(r_SC_M(1)*cosd(beta)+r_SC_M(3)*sind(beta)));
29
30%%% Z component of Moon-SC position vector
31r_M_SC(3) = (R_M/tand(theta/2))*(-r_SC_M(3)*cosd(beta) + r_SC_M(1)*sind(beta));
32
33%%% ECI position vector
34r_E_SC = r_E_M + r_M_SC;
35
36end
