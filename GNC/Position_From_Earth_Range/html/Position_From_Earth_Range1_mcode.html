1function [r_E_SC] = Position_From_Earth_Range(q_E, alpha, beta, theta)
 2%%% Calculates the spacecraft position vector in ECI.
 3%
 4% Inputs:   q_E     Spacecraft_Earth quaternion         (vector)
 5%           alpha   Horizontal quaternion error angle   (deg)
 6%           beta    Vertical quaternion error angle     (deg)
 7%           theta   Moon angular diameter               (deg)
 8%
 9% Author:   Cameron Maywood
10% Created:  9/29/2016
11% Modified: 9/29/2016
12%
13%%
14%%% Earth radius, km
15R_E = 6378.137
16
17%%% Spacecraft-Moon attitude unit vector
18r_SC_E = Quaternion_To_Attitude(q_E);
19
20%%% Initialize r_E_SC
21r_E_SC = zeros(1,3);
22
23%%% X component of ECI position vector
24r_E_SC(1) = (R_E/tand(theta/2))*(r_SC_E(2)*sind(alpha)-cosd(alpha)*(r_SC_E(1)*cosd(beta)+r_SC_E(3)*sind(beta)));
25
26%%% Y component of ECI position vector
27r_E_SC(2) = (-R_E/tand(theta/2))*(r_SC_E(2)*cosd(alpha)+sind(alpha)*(r_SC_E(1)*cosd(beta)+r_SC_E(3)*sind(beta)));
28
29%%% Z component of ECI position vector
30r_E_SC(3) = (R_E/tand(theta/2))*(-r_SC_E(3)*cosd(beta) + r_SC_E(1)*sind(beta));
31
32end
