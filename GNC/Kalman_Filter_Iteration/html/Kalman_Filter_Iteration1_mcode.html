1%% Kalman_Filter_Iteration Function
 2%  Calculates the state vector of the next measurement given the Kalman
 3%  filter initial conditions of the previous measurement.
 4%
 5%  Author:   Cameron Maywood
 6%  Created:  11/04/2016
 7%  Modified: 12/10/2016
 8%            __________________________________________________________ 
 9%  Inputs:  |   x_hat_0   |   Initial reference trajectory deviance    |
10%           |     phi     |   Next state transition matrix             |
11%           |     P_0     |   Initial covariance matrix                |
12%           |      Y      |   Next state observation                   |
13%           |    X_ref    |   Next spacecraft 3-body reference state   |
14%           |      R      |   Next state error covariance              |
15%           |_____________|____________________________________________|
16%  Outputs: |    X_est    |   New estimated spacecraft state vector    |
17%           |    x_hat    |   New reference trajectory deviation       |
18%           |      P      |   New covariance matrix                    |
19%           |      y      |   Residuals matrix                         |
20%           |_____________|____________________________________________|
21%
22function [X_est, x_hat, P, y] = Kalman_Filter_Iteration(x_hat_0, phi, P_0, Y, X_ref, R)
23%% Time Update
24
25x_bar = phi*x_hat_0;
26P_bar = phi*P_0*phi';
27
28%% Observation Deviation, Observation State Matrix, Gain Matrix
29
30y = Y - X_ref(1:3);
31H = [eye(3), zeros(3)];
32K = P_bar*H'*(H*P_bar*H' + R)^-1;
33
34%% Measurement Update
35
36x_hat = x_bar + K*(y - H*x_bar);
37P = (eye(6) - K*H)*P_bar*(eye(6) - K*H)' + K*R*K';
38
39%% Updated Estimate
40
41X_est = X_ref + x_hat;
42end
