1function [r_E_SC1, r_E_SC2] = Position_From_Angles_Slew(r_E_M, q_E, q_M, alpha, beta, vel, time)
 2%%% Calculates the spacecraft ECI position vector utilizing the angles 
 3%%% method. This method takes the slew rate of the spacecraft between
 4%%% pictures into account and outputs the positions at the times of both
 5%%% images.
 6%
 7%   Author:   Cameron Maywood
 8%   Created:  10/31/2016
 9%   Modified: 11/06/2016
10%             _________________________________________________________
11%   Inputs:  |    r_E_M    |   Moon ECI position vector                |
12%            |    q_E      |   Spacecraft-Earth quaternion             | 
13%            |    q_M      |   Spacecraft-Moon quaternion              |
14%            |    alpha    |   Horizontal quaternion error angle       |
15%            |    beta     |   Vertical quaternion error angle         |
16%            |    vel      |   Spacecraft ECI velocity vector          |
17%            |    time     |   Time between Earth and Moon images      |
18%            |_____________|___________________________________________|
19%   Outputs: |   r_E_SC1   |   First spacecraft ECI position vector    | 
20%            |   r_E_SC2   |   Second spacecraft ECI position vector   |
21%            |_____________|___________________________________________|
22
23%%
24
25% Quaternion error rotation matrix
26ROT_Q = [cosd(alpha)*cosd(beta),    -sind(alpha),   cosd(alpha)*sind(beta);
27         sind(alpha)*cosd(beta),     cosd(alpha),   sind(alpha)*sind(beta);
28         -sind(beta),                0,             cosd(beta)];
29
30% Spacecraft-Earth attitude unit vector
31r_SC_E = ROT_Q*Quaternion_To_Attitude(q_E);
32
33% Spacecraft-Moon attitude unit vector
34r_SC_M= ROT_Q*Quaternion_To_Attitude(q_M);
35
36% Angle between inteception of both spacecraft unit vectors
37theta_int = acosd(dot(r_SC_E, r_SC_M));
38
39% Earth-Moon distance
40d_E_M = norm(r_E_M);
41
42% Angle between Earth and spacecraft
43theta_M = acosd(dot(-r_E_M, -r_SC_M)/d_E_M);
44
45% ECI position vector to inteception point
46r_E_int = d_E_M*sind(theta_M)/sind(theta_int)*-r_SC_E;
47
48% Posiriont vector from spacecraft at first picture to spacecraft at second
49% picture
50r_SC1_SC2 = time*vel;
51
52% Angle between spacecraft transient position vector and Moon-Spacecraft
53% unit vector
54theta_SC2 = acosd(dot(-r_SC1_SC2,-r_SC_M)/norm(-r_SC1_SC2));
55
56% Position vector from spacecraft at first position to interception point
57r_SC1_int = norm(r_SC1_SC2)*sind(theta_SC2)/sind(theta_int)*-r_SC_E;
58
59% ECI position vectors to both spacecraft positions
60r_E_SC1 = r_E_int - r_SC1_int;
61r_E_SC2 = r_E_SC1 + r_SC1_SC2;
62
63end
